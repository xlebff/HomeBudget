        [ObservableProperty]
        private ObservableCollection<Currency> _currencies = [];

        [ObservableProperty]
        private Currency _operatingCurrency = new();

        [ObservableProperty]
        private bool _isBusy;

        [ObservableProperty]
        private string _busyText;


        public async Task LoadCurrenciesAsync()
        {
            await ExecuteAsync(async () =>
            {
                var currencies = await _context.GetAllAsync<Currency>();
                if (currencies is not null && currencies.Any())
                {
                    Currencies ??= [];

                    foreach(var currency in currencies)
                    {
                        Currencies.Add(currency);
                    }
                }
            }, "Fetching products...");
        }


        [RelayCommand]
        private void SetOperatingCurrency(Currency? currency) =>
            OperatingCurrency = currency ?? new();

        [RelayCommand]
        private async Task SaveCurrencyAsync()
        {
            if (OperatingCurrency is null)
                return;

            var (isValid, errorMessage) = OperatingCurrency.Validate();
            if (!isValid)
            {
                await Shell.Current.DisplayAlertAsync("Validation Error", errorMessage, "Ok");
                return;
            }

            var busyText = OperatingCurrency.Id == 0 ? "Creating currency..." : "Updating currency...";
            await ExecuteAsync(async () =>
            {
                if (OperatingCurrency.Id == 0)
                {
                    await _context.AddItemAsync<Currency>(OperatingCurrency);
                    Currencies.Add(OperatingCurrency);
                }
                else
                {
                    if (await _context.UpdateItemAsync<Currency>(OperatingCurrency))
                    {
                        var currencyCopy = OperatingCurrency.Clone();

                        var index = Currencies.IndexOf(OperatingCurrency);
                        Currencies.RemoveAt(index);

                        Currencies.Insert(index, currencyCopy);
                    }
                    else
                    {
                        await Shell.Current.DisplayAlertAsync("Error", "Curreny updation error", "Ok");
                        return;
                    }
                }
                SetOperatingCurrencyCommand.Execute(new());
            }, busyText);
        }

        [RelayCommand]
        private async Task DeleteCurrencyAsync(int id)
        {
            await ExecuteAsync(async () =>
            {
                if (await _context.DeleteItemByKeyAsync<Currency>(id))
                {
                    var currency = Currencies.FirstOrDefault(c => c.Id == id);
                    if (currency is not null)
                        Currencies.Remove(currency);
                }
                else
                {
                    await Shell.Current.DisplayAlertAsync("Delete Error", "Currency was not deleted", "Ok");
                }
            }, "Deleting product...");
        }
